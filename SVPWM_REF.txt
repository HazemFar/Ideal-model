.SUBCKT SVPWM_REF A B C AP BP CP



Ran A 0 100MEG
Rbn B 0 100MEG
Rcn C 0 100MEG


.func getmaximum(AN,BN,CN) {max(max(AN,BN),CN)}
.func getminimum(AN,BN,CN) {min(min(AN,BN),CN)}
Rcalcoffset calc_offset 0 0.5
Goffset  calc_offset 0 value={ getmaximum(V(A),V(B),V(C)) + getminimum(V(A),V(B),V(C))}




Bvap AP 0 v = V(A)+V(calc_offset)  ;voltage of the pole
Bvbp BP 0 v = V(B)+V(calc_offset)
Bvcp CP 0 v = V(C)+V(calc_offset)


.ENDS SVPWM_REF
